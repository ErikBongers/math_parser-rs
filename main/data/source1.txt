function angleCorrection(height1, height2) 
  {
  a1 = atan(height1);
  a2 = atan(height2);
  deltaA = a2 - a1;
  res = tan(deltaA);
  res.=height1;
  }
angleCorrection(.70cm, .75).mm; //should be 0.328mm
